CSS — focus ring (add to stylesheet)

/* Focus visible for all interactive game tiles */ .rc-actor:focus-visible, .rc-film:focus-visible, .rc-ct-tile:focus-visible, .rc-ct-item:focus-visible { outline: 2px solid var(--gold); outline-offset: 2px; border-radius: 12px; z-index: 10; } /* Ensure focus ring isn't clipped by containers */ #actorDeck, #filmDeck, #chainTrack { overflow: visible; } /* Optional: subtle scale on focus for tactile feel */ .rc-actor:focus-visible, .rc-film:focus-visible { transform: scale(1.03); }

JS — one-time enhancement (run after render)

// Make all game tiles keyboard accessible function enhanceTiles() { // Actors document.querySelectorAll('.rc-actor.rc-pick').forEach(el => { if (!el.hasAttribute('role')) { el.setAttribute('role', 'button'); el.setAttribute('tabindex', '0'); const name = el.querySelector('.rc-nm')?.textContent?.trim() || 'actor'; el.setAttribute('aria-label', `Choose ${name}`); // Enter/Space triggers onclick el.addEventListener('keydown', e => { if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); el.click(); } }); } }); // Films document.querySelectorAll('.rc-film.rc-pick, .rc-film[onclick]').forEach(el => { if (!el.hasAttribute('role')) { el.setAttribute('role', 'button'); el.setAttribute('tabindex', '0'); const name = el.querySelector('.rc-flbl')?.textContent?.trim() || 'film'; el.setAttribute('aria-label', `Choose ${name}`); el.addEventListener('keydown', e => { if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); el.click(); } }); } }); // Chain track tiles (start/target) document.querySelectorAll('.rc-ct-tile[onclick], .rc-ct-item[onclick]').forEach(el => { if (!el.hasAttribute('role')) { el.setAttribute('role', 'button'); el.setAttribute('tabindex', '0'); const label = el.querySelector('.rc-ct-fb')?.textContent?.trim() || 'film'; const type = el.classList.contains('rc-here') ? 'Start' : 'Target'; el.setAttribute('aria-label', `${type} film: ${label}`); el.addEventListener('keydown', e => { if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); el.click(); } }); } }); } // Call after initial render and after each newPuzzle() enhanceTiles();
